import time
from lubanvision_hardware.sysfs_pwm import ServoPwmConfig, SysfsPwmChannel

steps = [
    ("center", 1_500_000),
    ("low", 1_450_000),
    ("center", 1_500_000),
    ("high", 1_550_000),
    ("center", 1_500_000),
]
with SysfsPwmChannel(ServoPwmConfig()) as pwm:
    pwm.enable_center()
    for name, pulse_ns in steps:
        pwm.set_pulse_ns(pulse_ns)
        print(name, pulse_ns, flush=True)
        time.sleep(2.0)
PY
